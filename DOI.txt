               truck_no
               transptrcd
               transplpt

               trk_purpos

               spart

               vstel
               pp_entr_dt

          FOR ALL ENTRIES IN lt_config_t
          WHERE transplpt = lt_config_t-transplpt AND" lw_config-transplpt"
                vstel     = lt_config_t-vstel AND" lw_config-vstel"            "Index YTTSTX0001~ROU
